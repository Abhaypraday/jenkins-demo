pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                bat 'javac HelloWorld.java'
            }
        }
        stage('Run') {
            steps {
                bat 'java HelloWorld'
            }
        }
    }
    post {
        always {
            archiveArtifacts artifacts: '**/*.class', fingerprint: true
        }
    }
}
